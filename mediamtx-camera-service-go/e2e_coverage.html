/*
E2E Test Helpers - Following Integration Test Asserter Pattern

Provides E2E testing utilities following the proven integration test asserter pattern
from tests/integration/security_auth_test.go. Eliminates code duplication by leveraging
existing testutils infrastructure and supporting multiple WebSocket connections for
multi-user security testing scenarios.

Key Principles:
- Direct embedding of *testutils.UniversalTestSetup (no wrapper)
- Single testutils.SetupTest() call per test
- Use testutils.DefaultTestTimeout (30s) not 1s
- Support multiple connections via slice
- No custom tempDirs tracking
- No custom timeout constants
*/

package e2e

import (
        "context"
        "net/url"
        "testing"
        "time"

        "github.com/camerarecorder/mediamtx-camera-service-go/internal/security"
        "github.com/camerarecorder/mediamtx-camera-service-go/internal/testutils"
        "github.com/gorilla/websocket"
        "github.com/stretchr/testify/require"
)

// E2EAsserter follows integration test asserter pattern
type E2EAsserter struct {
        setup       *testutils.UniversalTestSetup
        jwtHandler  *security.JWTHandler
        connections []*websocket.Conn // Support multiple connections
}

// NewE2EAsserter creates asserter using existing patterns
func NewE2EAsserter(t *testing.T) *E2EAsserter {
        setup := testutils.SetupTest(t, "config_valid_complete.yaml")
        config := setup.GetConfigManager().GetConfig()
        jwtHandler, err := security.NewJWTHandler(config.Security.JWTSecretKey, setup.GetLogger())
        require.NoError(t, err, "Failed to create JWT handler")

        asserter := &E2EAsserter{
                setup:       setup,
                jwtHandler:  jwtHandler,
                connections: make([]*websocket.Conn, 0),
        }

        t.Cleanup(func() { asserter.Cleanup() })
        return asserter
}

// GenerateToken wraps JWT handler
func (a *E2EAsserter) GenerateToken(userID, role string, hours int) (string, error) {
        return a.jwtHandler.GenerateToken(userID, role, hours)
}

// ConnectWebSocket establishes and authenticates WebSocket connection
func (a *E2EAsserter) ConnectWebSocket(t *testing.T, token string) *websocket.Conn {
        ctx, cancel := context.WithTimeout(context.Background(), testutils.DefaultTestTimeout)
        defer cancel()

        u := url.URL{Scheme: "ws", Host: "localhost:8002", Path: "/ws"}
        conn, _, err := websocket.DefaultDialer.DialContext(ctx, u.String(), nil)
        require.NoError(t, err, "Failed to connect to WebSocket server")

        // Authenticate
        authResp := a.SendJSONRPC(t, conn, "authenticate", map[string]interface{}{
                "auth_token": token,
        })
        require.NotNil(t, authResp["result"], "Authentication should succeed")

        a.connections = append(a.connections, conn)
        return conn
}

// SendJSONRPC sends request and waits for response
func (a *E2EAsserter) SendJSONRPC(t *testing.T, conn *websocket.Conn, method string, params map[string]interface{}) map[string]interface{} {
        requestID := time.Now().UnixNano() % 1000000
        request := map[string]interface{}{
                "jsonrpc": "2.0",
                "method":  method,
                "params":  params,
                "id":      requestID,
        }

        err := conn.WriteJSON(request)
        require.NoError(t, err, "Failed to send JSON-RPC request")

        // Wait for response with timeout
        conn.SetReadDeadline(time.Now().Add(testutils.DefaultTestTimeout))
        var response map[string]interface{}
        err = conn.ReadJSON(&response)
        require.NoError(t, err, "Failed to read JSON-RPC response")

        return response
}

// Cleanup closes all connections
func (a *E2EAsserter) Cleanup() {
        for _, conn := range a.connections {
                if conn != nil {
                        conn.Close()
                }
        }
}
